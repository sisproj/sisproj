--글 읽은 여부 테이블
create table CF_IS_READ(
    SEQ_NO NUMBER PRIMARY KEY,
    EMP_NO NUMBER CONSTRAINT FK_READ_FROM_EMP REFERENCES EMP(EMP_NO) ON DELETE CASCADE,
    CF_NO VARCHAR2(100) CONSTRAINT FK_READ_FROM_CONFIRM REFERENCES CONFIRM(CF_NO) ON DELETE CASCADE,
    IS_READ VARCHAR2(10) DEFAULT 'N'
);

DROP TABLE CF_IS_READ CASCADE CONSTRAINTS; --드랍시키기

CREATE SEQUENCE CF_IS_READ_SEQ
START WITH 1
INCREMENT BY 1
NOMAXVALUE;
